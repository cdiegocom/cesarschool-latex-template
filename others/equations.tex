% equations.tex
\newcommand{\LzeroEquation}{
    \begin{equation}
        L_0(x) = \sum_{i} \mathbb{1}(x_i \neq 0)
        \label{eq:l0}
    \end{equation}
}

\newcommand{\LtwoEquation}{
    \begin{equation}
        L_2(x) = \left( \sum_{i} |x_i|^2 \right)^{\frac{1}{2}}
        \label{eq:l2}
    \end{equation}
}

\newcommand{\LinfEquation}{
    \begin{equation}
        L_\infty(x) = \max_i |x_i|
        \label{eq:linf}
    \end{equation}
}
